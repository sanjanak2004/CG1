#include <iostream>
#include <GL/glut.h>
#include <vector>
#include <stack>

const int WIDTH = 640;
const int HEIGHT = 480;

void setPixel(int x, int y, float r, float g, float b) {
    glColor3f(r, g, b);
    glBegin(GL_POINTS);
    glVertex2i(x, y);
    glEnd();
    glFlush();
}

void boundaryFill(int x, int y, float fillColorR, float fillColorG, float fillColorB, float borderColorR, float borderColorG, float borderColorB) {
    std::stack<std::pair<int, int>> pixels;
    pixels.push(std::make_pair(x, y));

    while (!pixels.empty()) {
        std::pair<int, int> current = pixels.top();
        pixels.pop();

        int cx = current.first;
        int cy = current.second;

        // Read the current pixel color
        float currentPixel[3];
        glReadPixels(cx, cy, 1, 1, GL_RGB, GL_FLOAT, currentPixel);

        if (currentPixel[0] != borderColorR || currentPixel[1] != borderColorG || currentPixel[2] != borderColorB) {
            // Set pixel to fill color
            setPixel(cx, cy, fillColorR, fillColorG, fillColorB);

            // Push neighboring pixels onto stack for further processing
            pixels.push(std::make_pair(cx + 1, cy));
            pixels.push(std::make_pair(cx - 1, cy));
            pixels.push(std::make_pair(cx, cy + 1));
            pixels.push(std::make_pair(cx, cy - 1));
        }
    }
}

void mouseCallback(int button, int state, int x, int y) {
    if (button == GLUT_LEFT_BUTTON && state == GLUT_DOWN) {
        y = HEIGHT - y; // Invert y-coordinate to match OpenGL's bottom-left origin
        boundaryFill(x, y, 1.0f, 0.0f, 0.0f, 1.0f, 1.0f, 1.0f); // Fill with red (1.0, 0.0, 0.0), border color white (1.0, 1.0, 1.0)
    }
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);

    glColor3f(0.0f, 0.0f, 0.0f); // Black color for polygon border
    glBegin(GL_LINE_LOOP);
    glVertex2i(200, 200);
    glVertex2i(400, 400);
    glVertex2i(200, 400);
    glEnd();

    glFlush();
}

void init() {
    glClearColor(1.0f, 1.0f, 1.0f, 1.0f); // White background color
    glMatrixMode(GL_PROJECTION);
    glLoadIdentity();
    gluOrtho2D(0, WIDTH, 0, HEIGHT);
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(WIDTH, HEIGHT);
    glutCreateWindow("Boundary Fill Polygon Filling");

    glutDisplayFunc(display);
    glutMouseFunc(mouseCallback);

    init();

    glutMainLoop();

    return 0;
}
